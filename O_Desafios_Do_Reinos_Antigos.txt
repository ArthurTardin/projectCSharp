using System;

class Program
{
    static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.UTF8;
        // Entrada de dados
        Console.WriteLine("=== Triagem da Guilda contra o Dragão Vermelho 🐉 ===");

        Console.Write("Nome do Herói: ");
        string nome = Console.ReadLine();

        Console.Write("Classe do Herói (Mago, Clérigo, Guerreiro, Druida, Anão, Elfo ou Troll): ");
        string classe = Console.ReadLine();

        Console.Write("Nível do Herói: ");
        int nivel = int.Parse(Console.ReadLine());

        Console.Write("Pontos de Vida (HP): ");
        double hp = double.Parse(Console.ReadLine());

        Console.Write("Pontos de Mana: ");
        int mana = int.Parse(Console.ReadLine());

        Console.Write("Ouro na mochila: ");
        double ouro = double.Parse(Console.ReadLine());

        Console.WriteLine();

        // REGRA 1 – Troll é expulso
        if (classe == "Troll")
        {
            Console.WriteLine("🚫 Troll detectado! A guilda expulsa imediatamente.");
            return;
        }
        else if (classe != "Troll" && !(nivel < 5))
        {
            Console.WriteLine("✅ Herói aceito na guilda!");
        }
        else
        {
            Console.WriteLine("❌ Nível muito baixo para entrar na guilda.");
            return;
        }

        // REGRA 2 – Cálculo do poder base
        double poderBase = (nivel * 10) + (hp / 2) - 5;
        Console.WriteLine($"⚔️ Poder Base do herói: {poderBase}");

        // REGRA 3 – Dano crítico
        if (poderBase % 2 == 0)
        {
            Console.WriteLine("✨ Ataque Crítico ativado! O herói possui vantagem!");
        }

        bool podeAtacar = false;

        // REGRA 4 – Magia
        if (classe == "Mago" || classe == "Druida" || classe == "Clérigo")
        {
            if (mana >= 30 && nivel <= 100)
            {
                Console.WriteLine("🧪 O herói conjura um poderoso feitiço contra o dragão!");
                podeAtacar = true;
            }
            else
            {
                Console.WriteLine("🪵 O cajado falhou! Mana insuficiente.");
            }
        }

        // REGRA 5 – Ataque físico
        else if (classe == "Guerreiro" || classe == "Anão" || classe == "Elfo")
        {
            if (hp > 20 && poderBase > 50)
            {
                Console.WriteLine("🛡️ O herói parte para um ataque físico contra o dragão!");
                podeAtacar = true;
            }
            else
            {
                Console.WriteLine("😬 O herói não tem força suficiente para atacar.");
            }
        }

        // REGRA 6 – Suborno
        if (!podeAtacar)
        {
            Console.WriteLine("💰 Tentando subornar o dragão...");

            if (ouro > 1000 || classe == "Anão")
            {
                Console.WriteLine("🐉 O dragão aceita o suborno e vai embora satisfeito!");
            }
            else
            {
                Console.WriteLine("🔥 O dragão não aceita o suborno...");
                Console.WriteLine("🍖 O herói virou churrasco de dragão.");
            }
        }
    }
}
